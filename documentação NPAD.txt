4. 	#!/bin/bash

	#SBATCH --time=0-0:5    # Especifica o tempo máximo de execução do job, dado no padrão dias-horas:minutos

5. chmod +x script_11.sh # para habilitar o script apos salvar o arquivo

6. ./script_11.sh #Para executar no nó de login apenas (como vinhamos fazendo)

7. sbatch script_11.sh #a saida apos este comando sera  Submitted batch job JobID
   #Onde JobID será substituído pelo número que identificará o job. A saída do programa não aparecerá na tela. Ela será escrita em um arquivo de nome slurm-JobID.out, 
   onde JobID será o mesmo valor do id do job submetido.

8. cat slurm-JobID.out  #apos este comando que veeremos o resultado do que foi processado
